\documentclass{article}
\usepackage{cheatsheet}
\title{1Password Cheat Sheet}
\author{Rebecca Turner}
\date{2022-05-15}
\begin{document}

\section{Global}
\begin{description}
  \bind{\shift-\cmd-Space}{Show quick access}
  \bind{\shift-\cmd-\backslash}{Show 1Password (custom)}
  \bind{\shift-\cmd-L}{Lock 1Password}
  \bind{\cmd-\backslash}
\end{description}

\section{Application}
\begin{description}
  \bind{\cmd-1}{Switch to all accounts}
  \bind*{\key{\cmd-2} through \key{\cmd-9}}{Switch accounts/collections}
  \bind{\cmd-[}{Back}
  \bind{\cmd-]}{Forward}
  \bind{\cmd-C}{Copy primary field}
  \bind{\shift-\cmd-C}{Copy password}
  \bind{\alt-\cmd-C}{Copy one-time password}
  \bind{\shift-\cmd-F}{Open and fill in web browser}
  \bind{\cmd-E}{Edit item}
  \bind{\cmd-S}{Save item}
  \bind{Backspace}{Archive item}
  \bind{\cmd-Backspace}{Delete item}
\end{description}

\end{document}
